状态机

三段式状态机思想
发布日期:   2021-10-19

状态机在FPGA设计中具有十分重要的地位，可以实现类似单片机一样的按一定流程“执行”。

状态机的定义

​ 状态机就是将一个复杂的问题分为很多个步骤，这一步做好了才能去到下一步。比如说实现一个i2c协议发送单字节数据过程，第一步是发送地址；第二步是等待应答，应答成功后进入第三部发送寄存器地址，失败则回到空闲态，然后再次等待发送信号；第四步也是等待应答，应带成功进入第五步发送一个字节数据，失败则返回空闲态，然后等待发送信号，从头开始；第六步等待应答，应答成功则发送stop信号，然后回到空闲态，等下一个发送信号，失败则回到空闲态等发送信号。当然，可以加入发送失败标志判断是否需要重新发送。

状态机的描述方法

类型 描述方法
一段式 整个状态机写到一个always模块里面，在该模块中既描述状态转移，又描述状态的输入和输出；
二段式 用两个always模块来描述状态机，其中一个always模块采用同步时序描述状态转移；另一个模块采用组合逻辑判断状态转移条件，描述状态转移规律以及输出；
三段式 在两个always模块描述方法基础上，使用三个always模块，一个always模块采用同步时序描述状态转移，一个always采用组合逻辑判断状态转移条件，描述状态转移规律，另一个always模块描述状态输出(可以用组合电路输出，也可以时序电路输出)。
二段式状态机的示例如下：
// 第一个进程，同步时序always模块，格式化描述次态寄存器迁移到现态寄存器
always @(posedge clk or negedge rst_n) begin
    if(!rst_n)
        curr_state <= INIT;
    else 
        curr_state <= next_state; // 注意，使用的是非阻塞赋值
end

//第二个进程，采用组合逻辑判断状态转移条件，描述状态转移规律以及输出
always @ (*) begin          //电平触发,组合逻辑
    case(current_state)
        S1: if(...)
                begin
                    data_out = ...;
                    next_state = S2; //阻塞赋值
                end
            else
                begin
                    data_out = ...;
                    next_state = S2; //阻塞赋值
                end
        ...:
    endcase
end
三段式状态机的示例如下：
// 第一个进程，同步时序always模块，格式化描述次态寄存器迁移到现态寄存器
always @(posedge clk or negedge rst_n) begin
    if(!rst_n)
        curr_state <= INIT;
    else 
        curr_state <= next_state;
end

//第二个进程，组合逻辑always模块，描述状态转移条件判断
always @ (*) begin          //电平触发,组合逻辑
    case(current_state)
        S1: if(...)
                begin
                    ...;             //一些条件的判断
                    next_state = S2; //阻塞赋值
                end
            else
                begin
                    ...;             //一些条件的判断
                    next_state = S2; //阻塞赋值
                end
        ...:
    endcase
end
            
            
always @(posedge clk or negedge rst_n) begin
    if(!rst_n)
        begin
            ...;                 //初始化
        end
    else
        begin
            case(next_state)
                S1:
                    out1 <= 1'b1; //注意是非阻塞逻辑
                S2:
                    out2 <= 1'b1;
                default:...;
            endcase
        end
end
两段式有限状态机与三段式有限状态机的区别

​ FSM将时序部分（状态转移部分）和组合部分（判断状态转移条件和产生输出）分开，写为两个always语句，即为两段式有限状态机。将组合部分中的判断状态转移条件和产生输入再分开写，则为三段式有限状态机。二段式在组合逻辑特别复杂时适用，但要注意需在后面加一个触发器以消除组合逻辑对输出产生的毛刺。三段式没有这个问题，由于第三个always会生成触发器。

自动售货机示例

​ 只列举了一种货物的情况：为6元的货物，只接受1元和5元的货币，当接受的钱足够时，会吐出货物。当在60个时钟周期内没投够6元时，自动回到初始状态，重新选择货物。货物每出货一个，总数减1，为0时重新装货，如此循环。

`timescale 1ns / 1ps

module shopping_1(
    input clk,
    input rst_n,
    input [2:0] din,                           // 投1元，5元
    input [2:0] choose,                        // 货物选择 1：6元，2：8元
    input confirm,                             // 选择后确认货物
    input reconfirm,                           // 确认后可以重新选择
    output reg dout,                           // 出货
    output reg [4:0] change,                   // 找钱
    output reg empty_6                         // 6元货物空信号
    );

reg [5:0] curr_state, next_state;
reg [4:0] count                 ;              // 当前总金额 
reg [6:0] goods_6               ;              // 初始63
reg [7:0] time_cnt              ;              // 计时60s
reg time_out                    ;              // 时间计数完信号
reg time_begin                  ;              // 时间开始计数信号
reg [4:0] temp                  ;              // 存放每个商品的价格，找钱用

// -----------------------货物全局初始化-------------------------
parameter INIT   = 11              ;              // 货物不够就初始化装货
parameter IDLE   = 22              ;              // 选择货物
// ------------------------------------------------------------

// -----------------------6元货物状态：-------------------------
parameter S6_0   = 1              ;              // 支付过程
parameter S6_1   = 2              ;              // 满足金额出货
// ------------------------------------------------------------


always @(posedge clk or negedge rst_n) begin
    if(!rst_n)
        curr_state <= INIT;
    else 
        curr_state <= next_state;
end

/**
 * @功能: 计时60s
 */
always @(posedge clk or negedge rst_n) begin        
    if(!rst_n)
        begin
            time_cnt <= 8'd0;
            time_out <= 8'd0;
        end
    else if(next_state == IDLE)
        begin
            time_cnt <= 8'd0;
            time_out <= 8'd0;
        end
    else
        begin
            if (time_begin == 1'd1)
                begin
                    if (time_cnt >= 8'd60)
                        begin
                            time_cnt <= 8'd0;
                            time_out <= 1'd1;
                        end
                    else 
                        begin
                            time_cnt <= time_cnt + 8'd1;
                            time_out <= 1'd0;
                        end
                end
            else
                begin
                    time_cnt <= 8'd0;
                    time_out <= 8'd0;
                end
        end
end

/**
 * @功能: 描述状态转移，和一些信号的产生。
 */
always @(*) begin
    case (curr_state)
        INIT:
            begin
                time_begin = 1'b0;
                goods_6 = 7'd10;
                next_state = IDLE;
                count = 5'd0;
            end
        IDLE:
            if(empty_6 == 1'b1)
                begin
                    time_begin = 1'b0;
                    goods_6 = 7'd10;
                    next_state = INIT;
                    count = 5'd0;
                end
            else
                begin
                    if(choose == 3'd1)
                        begin
                            time_begin = 1'b1;
                            goods_6 = goods_6;
                            next_state = S6_0;
                            count = 5'd0;
                        end 
                    else
                        begin
                            time_begin = 1'b0;
                            goods_6 = goods_6;
                            next_state = IDLE;
                            count = 5'd0;
                        end
                end
        S6_0:
            begin
                if (reconfirm == 1'b1)   // 是否重新选择
                    begin
                        time_begin = 0;
                        goods_6 = goods_6;
                        count = 5'd0;
                        next_state = IDLE;
                    end
                else 
                    begin
                        if (time_out == 1'b0)
                            begin
                                if (count >= 5'd6)
                                    begin
                                        time_begin = 1'b0;
                                        count = count;
                                        goods_6 = goods_6 - 7'd1;
                                        next_state = S6_1;
                                    end
                                else
                                    begin
                                        count = count + din;
                                        time_begin = time_begin;
                                        goods_6 = goods_6;
                                        next_state = S6_0; 
                                    end
                            end
                        else
                            begin
                                count = count;
                                goods_6 = goods_6;
                                time_begin = 1'b0;
                                next_state = IDLE;
                            end
                    end
            end
        S6_1:
            begin
                count = count;
                goods_6 = goods_6 - 7'd1;
                goods_6 = goods_6;
                next_state = IDLE;
            end
        default: ;
    endcase
end

always @(posedge clk or negedge rst_n) begin
    if(!rst_n)
        begin
            dout <= 1'b0;
            change <= 5'd0;
            empty_6 <= 1'b0;
        end
    else
        begin
            if(curr_state == S6_0 && time_out == 1'b1)
                begin
                    change <= count;
                    dout <= 1'b0;
                    empty_6 <= 1'b0;
                end
            else if(curr_state == S6_1)
                begin
                    change <= count - 5'd6;
                    dout <= 1'b1;
                    if(goods_6 <= 7'd0)
                        empty_6 <= 1'b1;
                    else
                        empty_6 <= 1'b0;
                end
            else
                begin
                    change <= 5'd0;
                    dout <= 1'b0;
                    empty_6 <= 1'b0;
                end
        end
end
endmodule

module tb_shopping_1;

// shopping_1 Parameters
parameter PERIOD = 10;
parameter INIT  = 11;
parameter IDLE  = 22;
parameter S6_0  = 1 ;
parameter S6_1  = 2 ;

// shopping_1 Inputs
reg   clk                                           = 0 ;
reg   rst_n                                         = 0 ;
reg   [2:0]  din                                    = 0 ;
reg   [2:0]  choose                                 = 0 ;
reg   confirm                                       = 0 ;
reg   reconfirm                                     = 0 ;
reg   [10:0] count ,data_out                        = 0 ;

// shopping_1 Outputs
wire  dout                                 ;
wire  [4:0]  change                        ;
wire  empty_6                              ;

integer i, fp, data, file_wr, fp1, data_choose, j;

initial
begin
    forever #(PERIOD/2)  clk=~clk;
end


initial begin
    #(PERIOD*2) rst_n  =  1;
    file_wr = $fopen("C:/Users/.../change.txt","w");
    fp = $fopen("C:/Users/.../money.txt", "r");
    fp1 = $fopen("C:/Users/.../choose.txt", "r");
    #(PERIOD*10000);
    $fclose(fp);
    $fclose(fp1);
end

always @ (posedge clk or negedge rst_n)begin
    
    if(!rst_n)begin
        choose <= 0;
        data_choose <= 0;
    end
    else begin
        for (i=0; fp!="\0";i=i+1) begin
            #(PERIOD);
            $fscanf(fp, "%d", data);
            din <= data;
        end
    end
end



always @ (posedge clk or negedge rst_n)begin
    
    if(!rst_n)begin
        choose <= 0;
        data_choose <= 0;
    end
    else begin
        for (j=0; fp1!="\0";j=j+1) begin
            #(PERIOD);
            $fscanf(fp1, "%d", data_choose);
            choose <= data_choose;
        end
    end
end


always @ (posedge clk or negedge rst_n)begin
    if(!rst_n)begin
        data_out<=8'b0;
        count<=11'b0;
        choose <= 0;
    end
    else if(count<11'd100) begin
        data_out<=change;
        count<=count+1'b1;
        $fdisplay(file_wr,"%d",change);
    end
    else 
        $fclose(file_wr);
end


shopping_1 #(
    .INIT ( INIT ),
    .IDLE ( IDLE ),
    .S6_0 ( S6_0 ),
    .S6_1 ( S6_1 ))
 u_shopping_1 (
    .clk                     ( clk              ),
    .rst_n                   ( rst_n            ),
    .din                     ( din        [2:0] ),
    .choose                  ( choose     [2:0] ),
    .confirm                 ( confirm          ),
    .reconfirm               ( reconfirm        ),

    .dout                    ( dout             ),
    .change                  ( change     [4:0] ),
    .empty_6                 ( empty_6          )
);

endmodule
